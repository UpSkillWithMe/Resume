Make a Game 'Catchme' using Python Tkinter

Introduction

Click here to see the short video of this tutorial. In this blog, Learn how to make a simple Catchme game using Python Tkinter. A Tkinter-based Python program that creates a game where the user needs to catch randomly changing shapes. The game window is set to be always on top and transparent. The shapes appear at random positions on the screen, and the user can catch them by clicking on them. The game continues until the user stops it or closes the window.

Full source code

import tkinter as tk
import random
import tkinter.messagebox

window = tk.Tk()
window.attributes('-transparentcolor', 'white')
window.overrideredirect(True)
window.attributes("-topmost", True) # set the window to be always on top
window.grab_set_global() # grab the focus globally

global count
count =0

shapes = [
    {'type': 'circle', 'coords': (100, 100, 300, 300), 'color': 'blue'},
    {'type': 'oval', 'coords': (100, 50, 300, 250), 'color': 'red'},
    {'type': 'hexagon', 'coords': (200, 50, 300, 100, 300, 200, 200, 250, 100, 200, 100, 100), 'color': 'green'},
    {'type': 'triangle', 'coords': (50, 100, 200, 300, 350, 100), 'color': 'orange'},
    {'type': 'parallelogram', 'coords': (100, 50, 300, 50, 250, 200, 50, 200), 'color': 'purple'},
    {'type': 'rhombus', 'coords': (200, 50, 350, 150, 200, 250, 50, 150), 'color': 'pink'},
    {'type': 'heart', 'coords': (150, 150, 250, 250), 'color': 'magenta'},
    {'type': 'amoeba', 'coords': (200, 50, 250, 100, 300, 100, 350, 150, 350, 200, 250, 250, 200, 250, 150, 200), 'color': 'yellow'}
]

is_game_active = True

def change_shape():
    if not is_game_active:
        return

    global shape
    shape = random.choice(shapes)
    window.geometry(f"{random.randint(200, 400)}x{random.randint(200, 400)}+{random.randint(0, window.winfo_screenwidth() - 400)}+{random.randint(0, window.winfo_screenheight() - 400)}")
    canvas.delete('current_shape')
    if shape['type'] == 'circle':
        canvas.create_oval(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'oval':
        canvas.create_oval(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'hexagon':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'triangle':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'parallelogram':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'rhombus':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'heart':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    elif shape['type'] == 'amoeba':
        canvas.create_polygon(shape['coords'], fill=shape['color'], tags='current_shape')
    global count 
    count+=1

    catch_me_text.config(bg=shape['color'])
    
    window.after(600, change_shape)  # Change shape every 1 second

def stop_game(event):
    window.attributes("-topmost", False)  # Remove the always on top attribute
    window.grab_release()  # Release the global focus
    tkinter.messagebox.showinfo("Congratulations.",  f"You caught upto {count} shape.")
    global is_game_active
    is_game_active = False
    window.destroy()

canvas = tk.Canvas(window, bg='white', highlightthickness=0)
canvas.pack(fill='both', expand=True)

catch_me_text = tk.Label(canvas, text="Catch me!", font=('Arial', 20))
catch_me_text.place(relx=0.5, rely=0.5, anchor=tk.CENTER)
canvas.bind('', stop_game)  # Bind left mouse click event to stop_game function

change_shape()
window.mainloop()
