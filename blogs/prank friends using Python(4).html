import tkinter as tk
import keyboard

def move_window():
    # Get the current mouse coordinates
    x, y = root.winfo_pointerxy()
    # Move the window to the mouse coordinates
    root.geometry(f"+{x}+{y}")
    # Schedule the next window movement
    root.after(1, move_window)  # Adjust the delay as needed

def disable_windows_key():
    for i in range(150):
        keyboard.block_key(i)

# Create the tkinter window
root = tk.Tk()
root.title("Moving Window")
root.geometry("200x200")

root.grab_set_global() #grab the focus globally

disable_windows_key()

# Start the window movement

# Start the tkinter event loop
