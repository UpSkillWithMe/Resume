Full source code

import winreg

def set_windows_theme(theme):
    # Define the Registry key paths
    key_path = r'SOFTWARE\Microsoft\Windows\CurrentVersion\Themes\Personalize'
    sub_key_path = r'SOFTWARE\Microsoft\Windows\CurrentVersion\Themes\Personalize\SystemUsesLightTheme'

    # Open the Registry key
    try:
        key = winreg.OpenKey(winreg.HKEY_CURRENT_USER, key_path, 0, winreg.KEY_WRITE)
    except FileNotFoundError:
        # The key does not exist, so create it
        key = winreg.CreateKey(winreg.HKEY_CURRENT_USER, key_path)

    # Set the theme value
    if theme == 'dark':
        winreg.SetValueEx(key, 'AppsUseLightTheme', 0, winreg.REG_DWORD, 0)
    elif theme == 'light':
        winreg.SetValueEx(key, 'AppsUseLightTheme', 0, winreg.REG_DWORD, 1)

    # Close the Registry key
winreg.CloseKey(key)

# Set the theme to dark
set_windows_theme('dark')

# Set the theme to light
set_windows_theme('light')
