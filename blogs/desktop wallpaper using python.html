Full source code

# change desktop wallpaper
import ctypes

def change_wallpaper(image_path):
    # Load the user32 library
    user32 = ctypes.windll.user32
    
    # Set the SPI constant for changing the desktop wallpaper
    SPI_SETDESKWALLPAPER = 0x0014
    
    # Call the SystemParametersInfo function to change the wallpaper
    result = user32.SystemParametersInfoW(SPI_SETDESKWALLPAPER, 0, image_path, 3)
    
    if result:
        print("Wallpaper changed successfully!")
    else:
        print("Failed to change wallpaper.")

# Usage
image_path = r"C:\Users\admin\Desktop\Resume\images\logo.png"  # Replace with the actual path of the image
change_wallpaper(image_path)
